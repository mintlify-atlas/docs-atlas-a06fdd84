## App Router Structure

The DragonHacks Event Platform uses Next.js 16's App Router for file-based routing. Routes are defined by the folder structure in `src/app/`.

### Route Map

```
src/app/
├── layout.tsx                    # Root layout (all pages)
├── page.tsx                      # / (Home/Landing)
├── providers.tsx                 # Client providers wrapper
├── globals.css                   # Global styles
│
├── discover/
│   └── page.tsx                  # /discover (Browse events)
│
├── create/
│   ├── page.tsx                  # /create (Create event form)
│   └── create-event-form.tsx     # Form component
│
├── events/
│   └── [id]/                     # /events/:id (Dynamic route)
│       ├── page.tsx              # Event detail page
│       ├── event-detail-content.tsx
│       ├── event-detail-registration.tsx
│       ├── event-detail-organizer-view.tsx
│       ├── event-detail-skeleton.tsx
│       ├── organizer-registrations.tsx
│       ├── attendance-qr-scanner.tsx
│       ├── announcements-section.tsx
│       ├── copy-link-button.tsx
│       └── settings/
│           ├── page.tsx          # /events/:id/settings
│           └── event-settings-form.tsx
│
├── signin/
│   ├── page.tsx                  # /signin (Email/Google sign-in)
│   ├── signin-form.tsx
│   └── phone/
│       ├── page.tsx              # /signin/phone
│       └── phone-signin-form.tsx

└── signup/
    ├── page.tsx                  # /signup (Email/Google sign-up)

        └── phone-signup-form.tsx
```

### Discover Page

**Route:** `/discover`  
**File:** `src/app/discover/page.tsx`

Browse and search events with filtering by category and title.

```tsx
// src/app/discover/page.tsx
import { DiscoverEvents } from "@/components/discover/discover-events";
import { DiscoverHeader } from "@/components/discover/discover-header";
import { EVENT_CATEGORIES, type EventCategory } from "@/types/events";

export default async function DiscoverPage({
  searchParams,
}: {
  searchParams: Promise<{ category?: string; title?: string }>;
}) {
  const params = await searchParams;
  const category = params.category as EventCategory | undefined;
  const title = params.title?.trim() || undefined;

  return (
    <div className="min-h-screen">
      <DiscoverHeader initialTitle={title} />

      <main className="mx-auto max-w-6xl px-8 py-8">
        <section className="mb-12">
          <h1 className="mb-3 text-4xl font-bold">Discover Events</h1>
          <p className="text-base text-zinc-500">
            Explore popular events and competitions, browse by category.
          </p>
        </section>

        <DiscoverEvents category={category} title={title} />
      </main>
    </div>
  );
}
```

**Search Params:**
- `?category=WORKSHOP` - Filter by event type
- `?title=hackathon` - Search by event name

### Authentication Routes

#### Email/Google Sign In

**Route:** `/signin`  
**File:** `src/app/signin/page.tsx`

```tsx
// src/app/signin/signin-form.tsx
"use client";

import { useState } from "react";
import { authClient } from "@/lib/auth-client";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";

export function SignInForm({ callbackUrl = "/discover" }: SignInFormProps) {
  const [email, setEmail] = useState("");
  const [password, setPassword] = useState("");

  const handleEmailSignIn = async (e: React.FormEvent) => {
    e.preventDefault();
    const { error } = await authClient.signIn.email({
      email,
      password,
      callbackURL: callbackUrl,
    });
    if (!error) {
      window.location.href = callbackUrl;
    }
  };

  const handleGoogleSignIn = () => {
    authClient.signIn.social({
      provider: "google",
      callbackURL: callbackUrl,
    });
  };

  return (
    <form onSubmit={handleEmailSignIn}>
      <Input type="email" placeholder="Email" value={email} onChange={(e) => setEmail(e.target.value)} />
      <Input type="password" placeholder="Password" value={password} onChange={(e) => setPassword(e.target.value)} />
      <Button type="submit">Sign in with email</Button>
      <Button type="button" variant="outline" onClick={handleGoogleSignIn}>
        Sign in with Google
      </Button>
    </form>
  );
}
```

## Navigation Patterns

### Next.js Link Component

Use Next.js `<Link>` for client-side navigation:

```tsx
import Link from "next/link";

<Link href="/discover">Browse Events</Link>
<Link href={`/events/${eventId}`}>View Event</Link>
```

### Programmatic Navigation

Use `useRouter` hook for navigation in client components:

```tsx
"use client";

import { useRouter } from "next/navigation";

export function MyComponent() {
  const router = useRouter();

  const handleSuccess = () => {
    router.push("/discover");
  };

  return <button onClick={handleSuccess}>Continue</button>;
}
```

### Callback URLs

Authentication routes support callback URLs for redirecting after sign-in:

```tsx
// Redirect to event page after sign-in
<Link href={`/signin?callbackUrl=${encodeURIComponent(`/events/${eventId}`)}`}>
  Sign in to register
</Link>
```

## Server vs Client Components

### Server Components (Default)

Pages are server components by default:
- `src/app/page.tsx` - Home page
- `src/app/discover/page.tsx` - Discover page (reads searchParams)
- `src/app/events/[id]/page.tsx` - Event detail page

### Client Components

Components requiring hooks or interactivity use `"use client"`:
- All form components (`*-form.tsx`)
- Authentication components (`HeaderAuth`, `OTPVerificationWizard`)
- Interactive UI (`DiscoverEvents`, `EventCard`)
- Provider wrappers (`Providers`)

```tsx
"use client";

import { useState } from "react";
import { useRouter } from "next/navigation";
// ... client-side logic
```

```tsx
// params.id = "abc123"

export default async function EventPage({
  params,
}: {
  params: Promise<{ id: string }>;
}) {
  const { id } = await params;
  // Fetch event with id
}
```

### Nested Dynamic Routes

Settings page nested under event:

```
/events/[id]/settings
```

Access the same `id` parameter in the nested route.
